procedure main {
	read a;
	read b;
	read c;
	call swap;
}

procedure swap {
	a = a + b;
	b = a - b;
	a = a - b;
	c  = a * (b - a);
	call auxiliary;
	b =  a + 5;
}

procedure auxiliary {
	read x;
	if (b < 15)  then {
		while (c <= x) {
			x = x + a; 
			c = 16;
		}
	} else {
		while (b >= c) {
			c = 7 * (a - b); 
			a = b + c;
		}
	}
	x = c;
	k = x + a - b + a * a; 
	print k;
}
